#include <stdio.h>
int main ()
{
    int a,b,c;
    printf("enter three sides of a tringle a b c");
    scanf("%d%d%d",&a,&b,&c);
    if(a==b&&b==c&&c==a)
    printf("it is an equilateral triangle");
    else if (a==b&&b==c&&c!=a)
    printf("it is an isosceles triangle");
    else if (a!=b&&b!=c&&c!=a)
    printf("it is a scalene triangle");
    else
    printf("not a valid triangle");
    return 0;
